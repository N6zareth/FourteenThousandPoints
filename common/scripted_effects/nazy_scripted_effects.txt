nazy_upgrade_building_sailors = {
	hidden_effect = {
		if = {
			limit = {
				has_building = dock
			}
			remove_building = dock
			add_building = drydock
		}
	}
	custom_tooltip = nazy_upgrade_building_sailors_tt
}
