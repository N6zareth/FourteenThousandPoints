    trigger = {	
    }
	
	mean_time_to_happen = {
	}
	
	immediate = {
		
	}
	
	option = {	#back
		name = bertus_limitation_menu.back
		highlight = yes
	}
	option = {	#all europe
		name = bertus_limitation_menu.51000.a
		hidden_effect = {
			every_province = {
				limit = { NOT = {continent = europe }}
				cede_province = ---
				reset_natives_colonizable_province = yes
				reset_native_ferocity_colonizable_province = yes
				reset_native_hostileness_colonizable_province = yes

			}
		}

	option = {	#
		name = bertus_limitation_menu.51000.b

			}
		}
	}
	option = {	#
		name = bertus_limitation_menu.51000.c
		hidden_effect = {
			every_province = {
				limit = { NOT = {continent = europe }}
				add_permanent_province_modifier = {
					name = unrule_lands
					duration = -1
				}
			}
		}
